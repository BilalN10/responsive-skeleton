@override
void onNavigation(NavigationResolver resolver, StackRouter router) {
  if (FirebaseAuth.instance.currentUser != null) {
    FirebaseAuth.instance.signOut().then(
      (value) {
        router.replaceNamed("/login");
      },
    );
  } else {
    router.replaceNamed("/login");
  }
}
